CREATE TABLE `log`.`rest` ( `id` INT NOT NULL AUTO_INCREMENT , `post_title` VARCHAR(100) NOT NULL , `post_detail` VARCHAR(1000) NOT NULL , `date` DATE NOT NULL DEFAULT CURRENT_TIMESTAMP , `time` TIME NOT NULL DEFAULT CURRENT_TIMESTAMP , `status` BOOLEAN NOT NULL DEFAULT TRUE , `img` VARCHAR(1000) NOT NULL , PRIMARY KEY (`id`)) ENGINE = InnoDB;
